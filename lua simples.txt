Inicio -> function main ap fp doispt Chunk end.
Chunk -> Stat Chunk | .
Block -> Chunk.
Stat -> while Exp do Block end | repeat Block until Exp | if Exp then Block Stat' G' end | Varlist igual Explist | 
print ap name fp | write ap name fp .
G' -> else Block | .
Stat' -> elseif Exp then Block Stat' | .
Varlist -> Var VL.
VL -> virg Var VL | .
Var -> name.
Explist -> Exp Explist'.
Explist' -> virg Exp Explist' | .
Exp -> false | true | EqNum.
Prefix_exp -> ap EqNum fp.
EqNum -> num EqNum' | Prefix_exp EqNum' | name EqNum'.
EqNum' -> Binop FBinop | .
FBinop -> num EqNum' | Prefix_exp EqNum' | name EqNum'.
Binop -> mais | menos | mul | div | sm | smi | sn | sni | igualdade | and | or | dif.
